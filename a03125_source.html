258 template <typename T, typename B> void QmlSharedPointer<T, B>::qt_static_metacall(QObject *_o, QMetaObject::Call _c, int _id, void **_a)
259 {
260  auto *_t = static_cast<QmlSharedPointer<T, B> *>(_o);
261  auto *_child = _t->data();
262  Q_ASSERT(_child != nullptr);
263 
264  //qDebug() << "Static metacall on..." << staticMetaObject.className() << _o << _c << _id;
265 
266  if (_c == QMetaObject::InvokeMetaMethod) {
267  auto newId = _id + staticMetaObject.superClass()->methodCount();
268 
269  auto method = QmlSharedPointer<T, B>::staticMetaObject.method(newId);
270  auto sig = QMetaObject::normalizedSignature(method.methodSignature());
271  if (method.methodType() == QMetaMethod::Signal) {
272  //qDebug() << "Firing signal" << sig;
273  QMetaObject::activate(_o, &staticMetaObject, _id, _a);
274  } else {
275  _child->qt_metacall(_c, newId, _a);
276  }
277  } else if (_c == QMetaObject::IndexOfMethod ||
278  _c == QMetaObject::ReadProperty ||
279  _c == QMetaObject::WriteProperty ||
280  _c == QMetaObject::ResetProperty) {
281  auto newId = _id + staticMetaObject.superClass()->propertyCount();
282  //qDebug() << "Access property" << newId;
283  _child->qt_metacall(_c, newId, _a);
284  } else {
285  Q_ASSERT(false); // Unhandled request
286  }
